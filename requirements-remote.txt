# Remote access (web / phone-style live voice / mobile) dependencies -- OPTIONAL.
#
# Kept separate from requirements.txt so the local voice assistant installs and
# runs without these. Install on the host that will serve remote clients:
#
#     pip install -r requirements-remote.txt
#
# Then run a LiveKit server (dev: `livekit-server --dev`, or LiveKit Cloud) and
# set LIVEKIT_URL / LIVEKIT_API_KEY / LIVEKIT_API_SECRET in the env / .env.
# Launch the worker with `python -m remote.worker` and serve the web client with
# `uvicorn remote.token_server:app`. See the remote/ + core/engines/livekit.py
# module docstrings for details.
#
# (STT/LLM/TTS still run locally via the base requirements; LiveKit is only the
# WebRTC transport, so fully-local is preserved.)

livekit                    # rtc primitives (Room/AudioSource/AudioStream/AudioFrame)
livekit-api                # AccessToken / VideoGrants (token server)
fastapi>=0.110             # token server + /chat + static web hosting
uvicorn>=0.29              # ASGI server for the token/web server
